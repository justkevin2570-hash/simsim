<div class="controls">
    <h2>단위원의 전개를 통한 사인함수 그래프</h2>
    <p>원의 둘레가 x축으로 펼쳐지며 $y = \sin(x)$를 그립니다.</p>
</div>
<canvas id="canvas"></canvas>

<script>
const canvas = document.getElementById('canvas');
const ctx = canvas.getContext('2d');

canvas.width = 1000;
canvas.height = 500;

const radius = 80;
const margin = 100;
const circleCenter = { x: 200, y: 250 };
const graphOrigin = { x: 400, y: 250 };

let angle = 0;
const speed = 0.02; // 애니메이션 속도
let points = []; // 그래프 좌표 저장

function draw() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);

    // 1. 좌표축 그리기
    ctx.strokeStyle = '#ccc';
    ctx.beginPath();
    // 원의 중심축
    ctx.moveTo(circleCenter.x - radius - 20, circleCenter.y); ctx.lineTo(circleCenter.x + radius + 20, circleCenter.y);
    ctx.moveTo(circleCenter.x, circleCenter.y - radius - 20); ctx.lineTo(circleCenter.x, circleCenter.y + radius + 20);
    // 그래프 축
    ctx.moveTo(graphOrigin.x, graphOrigin.y); ctx.lineTo(graphOrigin.x + radius * Math.PI * 2 + 50, graphOrigin.y);
    ctx.moveTo(graphOrigin.x, graphOrigin.y - radius - 20); ctx.lineTo(graphOrigin.x, graphOrigin.y + radius + 20);
    ctx.stroke();

    // 2. 단위원 그리기
    ctx.strokeStyle = '#333';
    ctx.lineWidth = 2;
    ctx.beginPath();
    ctx.arc(circleCenter.x, circleCenter.y, radius, 0, Math.PI * 2);
    ctx.stroke();

    // 3. 회전하는 점 계산 (y축 반전 고려하여 -angle)
    const px = circleCenter.x + radius * Math.cos(-angle);
    const py = circleCenter.y + radius * Math.sin(-angle);

    // 4. 원 안의 반지름 선과 점
    ctx.fillStyle = '#ff4757';
    ctx.beginPath();
    ctx.moveTo(circleCenter.x, circleCenter.y);
    ctx.lineTo(px, py);
    ctx.stroke();
    ctx.beginPath();
    ctx.arc(px, py, 5, 0, Math.PI * 2);
    ctx.fill();

    // 5. 사인 그래프 데이터 업데이트
    // x좌표는 각도(호의 길이)에 비례해서 우측으로 이동
    const gx = graphOrigin.x + angle * radius;
    const gy = py;
    points.push({ x: gx, y: gy });

    // 6. 연결선 (원 위의 점 -> 그래프 위의 점)
    ctx.setLineDash([5, 5]);
    ctx.strokeStyle = 'rgba(255, 71, 87, 0.5)';
    ctx.beginPath();
    ctx.moveTo(px, py);
    ctx.lineTo(gx, gy);
    ctx.stroke();
    ctx.setLineDash([]);

    // 7. 사인 곡선 그리기
    ctx.strokeStyle = '#1e90ff';
    ctx.lineWidth = 3;
    ctx.beginPath();
    points.forEach((p, i) => {
        if (i === 0) ctx.moveTo(p.x, p.y);
        else ctx.lineTo(p.x, p.y);
    });
    ctx.stroke();

    // 8. 바닥에 펼쳐지는 선 (원의 둘레가 펴지는 효과)
    ctx.strokeStyle = '#2ed573';
    ctx.lineWidth = 4;
    ctx.beginPath();
    ctx.moveTo(graphOrigin.x, graphOrigin.y + 2); // 살짝 아래에 표시
    ctx.lineTo(gx, graphOrigin.y + 2);
    ctx.stroke();

    // 애니메이션 제어: 2파이까지만 진행 후 초기화
    angle += speed;
    if (angle > Math.PI * 2) {
        angle = 0;
        points = [];
    }

    requestAnimationFrame(draw);
}

draw();
</script>
